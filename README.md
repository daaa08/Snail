#### 1차원 배열

- 타입 배열명[][] = new 타입[개수];
=> int array[] = new int[5];  (5개의 공간을 사용)
- 값 넣는 법
=> array[0] = 5;    //(5, , , , )
=> array[3] = 7;    //(5, , ,7, )
- 사용
=> System.out.println(array[3]); // 네번째 값 출력

#### 다차원 배열

- 타입 배열명[][][][] = new 타입[개수][개수][개수][개수];
  // []는 차원의 개수 만큼 사용, 차원은 맨 뒤가 1차원..[2차원][1차원]


#### 2차원 배열

=> int array[][] = new int[3][3];  // 3*3 공간 생성
// 1차원=가로축, 2차원= 세로축
- 값 넣는 법
=> array[0][0] = 5;  
=> array[0][1] = 7;  ...  
//  _0_  _1_  _2_
// _0_ **5**
// _1_ **7**
// _2_
